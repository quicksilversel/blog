success-notification:
  runs-on: ubuntu-latest
  steps:
    - name: Comment to PR
      uses: marocchino/sticky-pull-request-comment@v2
      with:
        header: success-notification
        message: |
          :tada: The Release was successful!
          You can view the changes.
